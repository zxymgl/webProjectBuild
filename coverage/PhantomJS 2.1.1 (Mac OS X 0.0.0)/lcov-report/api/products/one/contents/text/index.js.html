import 'regenerator-runtime'
import fetch from 'api/fetch'
import ParameterException from 'api/ParameterException'
 
export default class ContentText {
  constructor (baseUrl) {
    this.url = baseUrl + ''
  }
 
  url = ''
 
  async get (productId) {
    return await fetch({ method: 'GET', url: this.url.replace('{productId}', productId) })
  }
 
  async create (productId, data) {
    if (!data['productContentTypeEnumId']) throw new ParameterException('Product content data must has productContentTypeEnumId')
    if (!data['contentText']) throw new ParameterException('Product content data must has contentText')
 
    let formData = {
      productContentTypeEnumId: data['productContentTypeEnumId'],
      contentType: 'text',
      contentText: data['contentText']
    }
 
    if (data['locale']) formData['locale'] = data['locale']
    if (data['fromDate']) formData['fromDate'] = data['fromDate']
    if (data['thruDate']) formData['thruDate'] = data['thruDate']
    if (data['description']) formData['description'] = data['description']
 
    return await fetch({ method: 'POST', url: this.url.replace('{productId}', productId), body: formData })
  }
}
