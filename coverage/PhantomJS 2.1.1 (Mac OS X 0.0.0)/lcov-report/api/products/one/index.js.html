import 'regenerator-runtime'
import fetch from 'api/fetch'
 
import ProductCategories from './categories'
import ProductContents from './contents'
import ProductPrices from './prices'
import ProductFeatures from './features'
import ProductGeos from './geos'
import ProductStores from './stores'
 
export default class Product {
  constructor (baseUrl) {
    this.url = baseUrl + '/{productId}'
    this.categories = new ProductCategories(this.url)
    this.contents = new ProductContents(this.url)
    this.prices = new ProductPrices(this.url)
    this.features = new ProductFeatures(this.url)
    this.geos = new ProductGeos(this.url)
    this.stores = new ProductStores(this.url)
  }
 
  url = ''
  categories = null
  contents = null
  prices = null
  features = null
  geos = null
  stores = null
 
  async get (productId) {
    return await fetch({ method: 'GET', url: this.url.replace('{productId}', productId) })
  }
 
  async delete (productId) {
    return await fetch({ method: 'DELETE', url: this.url.replace('{productId}', productId) })
  }
}
