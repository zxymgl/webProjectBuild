import 'regenerator-runtime'
import fetch from 'api/fetch'
 
export default class ProductReviews {
  constructor (baseUrl) {
    this.url = baseUrl + '/reviews'
  }
 
  url = ''
 
  async get (productId) {
    return await fetch({ method: 'GET', url: this.url.replace('{productId}', productId) })
  }
 
  async create (productId, data) {
    return await fetch({ method: 'POST', url: this.url.replace('{productId}', productId), body: data })
  }
}
