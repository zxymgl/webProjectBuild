import 'regenerator-runtime'
import fetch from 'api/fetch'
import config from 'api/config'
 
import Product from './one'
import ProductCategories from './categories'
import ProductContents from './contents'
import ProductFeatures from './features'
import ProductGeos from './geos'
import ProductReviews from './reviews'
 
export default class Products {
  constructor (baseUrl) {
    this.url = baseUrl + '/products'
    this.one = new Product(this.url)
    this.categories = new ProductCategories(this.url)
    this.features = new ProductFeatures(this.url)
    this.geos = new ProductGeos(this.url)
    this.reviews = new ProductReviews(this.url)
    this.contents = new ProductContents(this.url)
  }
 
  url = ''
  one = null
  categories = null
  contents = null
  features = null
  geos = null
  reviews = null
 
  async get (pagination) {
    return await fetch({ method: 'GET', url: this.url,
      pagination: pagination !== undefined ? pagination : { pageIndex: 0, pageSize: config.pageSize } })
  }
 
  async create (data) {
    return await fetch({ method: 'POST', url: this.url, body: data })
  }
 
}
