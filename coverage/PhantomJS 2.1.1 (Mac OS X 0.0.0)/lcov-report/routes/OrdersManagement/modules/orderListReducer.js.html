import Immutable, { Record, List } from 'immutable'
import apiFactory from 'api/apiFactory'
const api = apiFactory()
import listData from './OrderListData'
/**
 * 获取服务商订单管理中心我的订单列表
 * @type {string}
 */
export const GET_ORDER_LIST_REQUEST = 'GET_ORDER_LIST_REQUEST'
export const GET_ORDER_LIST_SUCCESS = 'GET_ORDER_LIST_SUCCESS'
export const GET_ORDER_LIST_FAILURE = 'GET_ORDER_LIST_FAILURE'
/**
 * 获取服务商订单管理中心我的订单列表函数
 * @returns {{type: string}}
 */
export function getOrderListRequest () {
  return {type: GET_ORDER_LIST_REQUEST}
}
export function getOrderListSuccess (json) {
  return {
    type: GET_ORDER_LIST_SUCCESS,
    payload: json
  }
}
export function getOrderListFailure (error) {
  return {
    type: GET_ORDER_LIST_FAILURE,
    payload: error
  }
}
export function getOrderListFromServer () {
  return dispatch => {
    dispatch(getOrderListRequest())
    return api.orders.get()
      .then(json => {
        dispatch(getOrderListSuccess(json))
      }).catch(error => {
        dispatch(getOrderListFailure(error))
      })
  }
}
 
export const actions = {
  getOrderListFromServer
}
let InitialState = new Record({
  isLoading: false,
  ordersListData: new List()
})
const initialState = new InitialState()
export default function OrderListReducer (state = initialState, action) {
  if (!(state instanceof InitialState)) return initialState.merge(state)
  switch (action.type) {
    case GET_ORDER_LIST_REQUEST:
      return state.set('isLoading', false)
    case GET_ORDER_LIST_SUCCESS:
      var data = Immutable.fromJS(listData)
      return state.set('isLoading', false).set('ordersListData', data)
    case GET_ORDER_LIST_FAILURE:
      return state.set('isLoading', false)
  }
  return state
}
