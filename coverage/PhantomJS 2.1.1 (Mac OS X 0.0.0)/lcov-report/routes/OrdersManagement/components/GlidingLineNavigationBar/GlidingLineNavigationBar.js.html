import React from 'react'
import { Link } from 'react-router'
import styles from './GlidingLineNavigationBar.scss'
class GlidingLineNavigationBar extends React.Component {
  constructor (props) {
    super(props)
    this.state = {
      SwitchMenuIndex: 1
    }
  }
  static propTypes = {
    switchMenuIndex: React.PropTypes.num
  }
  render () {
    return (
      <div className='col-md-12'>
        <div className='col-lg-12 border-bottom'>
          <ul className='nav navbar-nav '>
            <li className={this.state.SwitchMenuIndex === 1 && styles.active} onClick={() => {this._handleSwitchMenuIndex(1) }}>
              <Link to='/ordersManagement/AllOrders' onClick={this._onCheckout.bind(this, 'one')}>
                <h4>全部订单</h4>
              </Link>
            </li>
            <li className={this.state.SwitchMenuIndex === 2 && styles.active} onClick={() => {this._handleSwitchMenuIndex(2) }}>
              <Link to='/ordersManagement/WaitingForPaymentOrders' onClick={this._onCheckout.bind(this, 'two')}>
                <h4>待付款</h4>
              </Link>
            </li>
            <li className={this.state.SwitchMenuIndex === 3 && styles.active} onClick={() => {this._handleSwitchMenuIndex(3) }}>
              <Link to='/ordersManagement/WaitingForAcceptanceOrders' onClick={this._onCheckout.bind(this, 'three')}>
                <h4>待受理</h4>
              </Link>
            </li>
            <li className={this.state.SwitchMenuIndex === 4 && styles.active} onClick={() => {this._handleSwitchMenuIndex(4) }}>
              <Link to='/ordersManagement/AlreadyAcceptedOrders' onClick={this._onCheckout.bind(this, 'four')}>
                <h4>已受理</h4>
              </Link>
            </li>
            <li className={this.state.SwitchMenuIndex === 5 && styles.active} onClick={() => {this._handleSwitchMenuIndex(5) }}>
              <Link to='/ordersManagement/NeedEvaluationOrders' onClick={this._onCheckout.bind(this, 'five')}>
                <h4>需要评价</h4>
              </Link>
            </li>
            <li className={this.state.SwitchMenuIndex === 6 && styles.active} onClick={() => {this._handleSwitchMenuIndex(6) }}>
              <Link to='/ordersManagement/SucceededOrders' onClick={this._onCheckout.bind(this, 'six')}>
                <h4>已成功</h4>
              </Link>
            </li>
          </ul>
        </div>
      </div>
    )
  }
  _handleSwitchMenuIndex (menuIndex) {
    this.setState({
      SwitchMenuIndex: menuIndex
    })
  }
}
export default GlidingLineNavigationBar
