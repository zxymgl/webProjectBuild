import React, { PropTypes } from 'react'
 
import StarRating from 'routes/Order/OrderPlacing/components/Score/components/StarRating'
class ServiceReviewStars extends React.Component {
  constructor (props) {
    super(props)
    this.state = {
    }
    this._renderStart = this._renderStart.bind(this)
  }
  static propTypes = {
    startLeavel: PropTypes.string
  }
  _renderStart = (startCount, cstarKind) => {
    var doms = []
    for (let i = 0; i < startCount; i++) {
      doms.push(<StarRating key={i + 1} isLight={true}  startLeavel={i + 1} />)
    }
    for (let i = 0; i < (5 - startCount); i++) {
      doms.push(<StarRating key={startCount + i + 1} isLight={false} startLeavel={startCount + i + 1} />)
    }
    return doms
  }
  render () {
    return (
      <div>
        {this._renderStart(this.props.startLeavel)}
        <span><small>{this.props.startLeavel}分</small></span>
      </div>
      )
  }
}
export default ServiceReviewStars
