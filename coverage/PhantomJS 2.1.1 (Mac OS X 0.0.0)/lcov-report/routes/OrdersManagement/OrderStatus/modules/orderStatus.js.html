import Immutable, { Record, List } from 'immutable'
import apiFactory from 'api/apiFactory'
const api = apiFactory()
import orderStatusData from './OrderStatusData'
/**
 * 获取订单状态信息
 * @type {string}
 */
export const GET_ORDER_STATUS_REQUEST = 'GET_ORDER_STATUS_REQUEST'
export const GET_ORDER_STATUS_SUCCESS = 'GET_ORDER_STATUS_SUCCESS'
export const GET_ORDER_STATUS_FAILURE = 'GET_ORDER_STATUS_FAILURE'
/**
 * 获取订单状态信息函数
 * @returns {{type: string}}
 */
export function getOrderStatusRequest () {
  return {type: GET_ORDER_STATUS_REQUEST}
}
export function getOrderStatusSuccess (json) {
  return {
    type: GET_ORDER_STATUS_SUCCESS,
    payload: json
  }
}
export function getOrderStatusFailure (error) {
  return {
    type: GET_ORDER_STATUS_FAILURE,
    payload: error
  }
}
export function getOrderStatusFromServer () {
  return dispatch => {
    dispatch(getOrderStatusRequest())
    /* return api.orders.addOne('DUMMY_ID') */ // TODO
    return api.products.categories.getProductCategoryList('AVATAR')
      .then(json => {
        dispatch(getOrderStatusSuccess(json))
      }).catch(error => {
        dispatch(getOrderStatusFailure(error))
      })
  }
}
 
export const actions = {
  getOrderStatusFromServer
}
let InitialState = new Record({
  isLoading: false,
  orderStatusData: new List()
})
const initialState = new InitialState()
export default function OrderStatusReducer (state = initialState, action) {
  if (!(state instanceof InitialState)) return initialState.merge(state)
  switch (action.type) {
    case GET_ORDER_STATUS_REQUEST:
      return state.set('isLoading', false)
    case GET_ORDER_STATUS_SUCCESS:
      var order = Immutable.fromJS(orderStatusData)
      return state.set('isLoading', false).set('orderStatusData', order)
    case GET_ORDER_STATUS_FAILURE:
      return state.set('isLoading', false)
  }
  return state
}
