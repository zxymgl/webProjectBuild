import React from 'react'
import { connect } from 'react-redux'
import { bindActionCreators } from 'redux'
import { Map } from 'immutable'
import OrderProgress from '../components/OrderProgress'
import OrderProfile from '../components/OrderProfile'
import OrdererProfile from '../components/OrdererProfile'
import styles from './OrderStatus.scss'
import { actions as getOrderStatusActions } from '../modules/orderStatus'
const actions = [
  getOrderStatusActions
]
function mapStateToProps (state) {
  return {
    statusData: state.orderstatus.orderStatusData
  }
}
function mapDispatchToProps (dispatch) {
  const creators = Map().merge(...actions).filter(value => typeof value === 'function').toObject()
  return {
    actions: bindActionCreators(creators, dispatch),
    dispatch
  }
}
 
class OrderStatus extends React.Component {
  render () {
    return (
      <div className={styles.background}>
        <OrderProgress data={this.props.statusData} />
        <br />
        <OrderProfile data={this.props.statusData} />
        <br />
        <OrdererProfile data={this.props.statusData} />
      </div>
    )
  }
  componentDidMount () {
    this.props.actions.getOrderStatusFromServer()
  }
}
OrderStatus.propTypes = {
  actions: React.PropTypes.object.isRequired,
  statusData: React.PropTypes.object
}
export default connect(mapStateToProps, mapDispatchToProps)(OrderStatus)
