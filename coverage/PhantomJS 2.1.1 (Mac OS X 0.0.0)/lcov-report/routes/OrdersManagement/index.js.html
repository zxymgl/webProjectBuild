import { injectReducer } from '../../store/reducers'
import createOrderListRoutes from './routes/index'
export default (store) => {
  let childRoute = createOrderListRoutes(store)
  return {
    path: 'OrdersManagement',
    /* 异步加载路由视图 Async getComponent is only invoked when route matches   */
    getComponent (nextState, cb) {
      /* 代码拆分 Webpack - use 'require.ensure' to create a split point
       and embed an async module loader (jsonp) when bundling   */
      require.ensure([], (require) => {
        /*  Webpack - use require callback to define
         dependencies for bundling   */
        const Order = require('./containers/OrdersManagement').default
        const reducer = require('./modules/orderListReducer').default
        /*  Add the reducer to the store on key 'counter'  */
        injectReducer(store, { key: 'orderstManagement', reducer })
        /*  Return getComponent   */
        cb(null, Order)
        /* Webpack named bundle   */
      }, 'OrdersManagement')
    },
    childRoutes: childRoute.childRoutes
  }
}
