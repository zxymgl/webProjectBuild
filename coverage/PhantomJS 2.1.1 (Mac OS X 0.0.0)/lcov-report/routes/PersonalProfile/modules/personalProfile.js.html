import Immutable, { Record, List } from 'immutable'
import apiFactory from 'api/apiFactory'
const api = apiFactory()
import personalProfileData from './PersonalData'
/**
 * 获取个人资料
 * @type {string}
 */
export const GET_PERSONAL_PROFILE_REQUEST = 'GET_PERSONAL_PROFILE_REQUEST'
export const GET_PERSONAL_PROFILE_SUCCESS = 'GET_PERSONAL_PROFILE_SUCCESS'
export const GET_PERSONAL_PROFILE_FAILURE = 'GET_PERSONAL_PROFILE_FAILURE'
/**
 * 获取个人资料函数
 * @returns {{type: string}}
 */
export function getPersonalProfileRequest () {
  return {type: GET_PERSONAL_PROFILE_REQUEST}
}
export function getPersonalProfileSuccess (json) {
  return {
    type: GET_PERSONAL_PROFILE_SUCCESS,
    payload: json
  }
}
export function getPersonalProfileFailure (error) {
  return {
    type: GET_PERSONAL_PROFILE_FAILURE,
    payload: error
  }
}
export function getPersonalProfile () {
  return dispatch => {
    dispatch(getPersonalProfileRequest())
    return api.products.categories.getProductCategoryList('CTX2000')
      .then(json => {
        dispatch(getPersonalProfileSuccess(json))
      }).catch(error => {
        dispatch(getPersonalProfileFailure(error))
      })
  }
}

export const actions = {
  getPersonalProfile,
  getCompanyProfile
}
let InitialState = new Record({
  isFetching: false,
  personalProfile: new List(),
  isLoading: false,
  companyProfile: new List()
})
const initialState = new InitialState()
export default function PersonalProfileReducer (state = initialState, action) {
  if (!(state instanceof InitialState)) return initialState.merge(action)
  switch (action.type) {
    case GET_PERSONAL_PROFILE_REQUEST:
      return state.set('isFetching', false)
    case GET_PERSONAL_PROFILE_SUCCESS:
      var persoanlData = Immutable.fromJS(personalProfileData)
      return state.set('isFetching', false).set('personalProfile', persoanlData)
    case GET_PERSONAL_PROFILE_FAILURE:
      return state.set('isFetching', false)
    case GET_COMPANY_PROFILE_REQUEST:
      return state.set('isLoading', false)
    case GET_COMPANY_PROFILE_SUCCESS:
      var personData = Immutable.fromJS(personalProfileData)
      return state.set('isLoading', false).set('companyProfile', personData)
    case GET_COMPANY_PROFILE_FAILURE:
      return state.set('isLoading', false)
  }
  return state
}
