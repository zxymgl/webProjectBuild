import React from 'react'
import ReactMenuAim from 'components/ReactMenuAim'
import classNames from 'classnames'
import styles from '../../containers/ServiceDetail.scss'
import UN from '../../assets/u71.png'
import PAY from '../../assets/u67.png'
import WEI from '../../assets/u69.png'
 
let storeNumber = 6
class OptionsAndQuantitySelection extends ReactMenuAim {
  constructor (props) {
    super(props)
    this.state = {
      number: 1
    }
    this._bind('_numAdd', '_buy')
  }
  _bind (...methods) {
    methods.forEach((method) => {
      this[method] = this[method].bind(this)
    })
  }
  render () {
    return (
      <div>
        {this.props.data.map(function (cp, index) {
          return (
            <div key={index}>
              <div className={classNames(styles.margintop, 'row col-md-12')}>
                <div className='col-md-3'><h3>数量</h3></div>
                <div className={classNames(styles.textalign_center, 'col-md-6')}>
                  <span className={classNames('label label-default', styles.btbg)}>-</span>
                  <input className={styles.buyinput} disabled='disabled' ref='thenumber' type='text' value={1} />
                  <span className={classNames('label label-default', styles.btbg)} >+</span>
                </div>
                <div className='col-md-3'><span className='pull-right'><h3>库存({cp.get('repertory')})</h3></span></div>
              </div>
              <div className={classNames(styles.margintop, 'row col-md-12')}>
                <div className={classNames(styles.textalign_center, 'col-md-8')}>
                  <span>
                    <input type='submit' className={styles.consider} value='立即购买' />
                  </span>
                </div>
                <div className='col-md-4'></div>
              </div>
              <div className={classNames(styles.margintop, 'row col-md-12')}>
                <div className='col-md-3'><span><h3>支付</h3></span></div>
                <div className='col-md-3'><img src={PAY} className={classNames(styles.littlebg)} /></div>
                <div className='col-md-3'><img src={WEI} className={classNames(styles.littlebg)} /></div>
                <div className='col-md-3'><img src={UN} className={classNames(styles.littlebg)} /></div>
              </div>
            </div>
          )
        })}
      </div>
    )
  }
  _numAdd () {
    var nuo = this.refs.thenumber.value.trim()
    if (nuo > storeNumber) {
      alert('购买数量不能超过库存!')
      return
    }
    nuo = parseInt(nuo) + 1
    this.setState({
      number: nuo
    })
  }
  _numDec (index) {
    var nuo = this.refs.thenumber.value.trim()
    if (parseInt(nuo) === 0) {
      alert('购买数量不能小于0')
      return
    }
    nuo = parseInt(nuo) - 1
    this.setState({
      number: nuo
    })
  }
  _buy () {
    alert('立即购买')
  }
}
OptionsAndQuantitySelection.propTypes = {
  data: React.PropTypes.object,
  number: React.PropTypes.number,
  _numAdd: React.PropTypes.func,
  _numDec: React.PropTypes.func,
  _buy: React.PropTypes.func
}
export default OptionsAndQuantitySelection
