import React from 'react'
import { bindActionCreators } from 'redux'
import { connect } from 'react-redux'
import { Map } from 'immutable'
import Tabs from 'react-bootstrap/lib/Tabs'
import Tab from 'react-bootstrap/lib/Tab'
import CumulativeReviews from '../../components/Reviews/CumulativeReviews'
import CompanyProfile from '../../components/CompanyProfile'
import ProductIntroduction from '../../components/ProductIntroduction'
import ServiceTerms from '../../components/ServiceTerms'
import classNames from 'classnames'
import styles from '../../containers/ServiceDetail.scss'
import { actions as serviceReviewActions } from '../../modules/cumulativeReviews'
import { actions as getBackgroundActions } from '../../modules/companyProfile'
const actions = [
  serviceReviewActions,
  getBackgroundActions
]
function mapStateToProps (state) {
  return {
    cumulativeReviews: state.servicedetail.cumulativeReviewsList,
    profile: state.servicedetail.checkout
  }
}
function mapDispatchToProps (dispatch) {
  const creators = Map().merge(...actions).filter(value => typeof value === 'function').toObject()
  return {
    actions: bindActionCreators(creators, dispatch),
    dispatch
  }
}
class DetailedProductDescription extends React.Component {
  constructor (props) {
    super(props)
    this.state = {
      type: 'one'
    }
    this._bind('_onSelect1', '_onSelect2', '_onSelect3')
  }
  _bind (...methods) {
    methods.forEach((method) => {
      this[method] = this[method].bind(this)
    })
  }
  render () {
    var content = <div></div>
    if (this.state.type === 'one') {
      content = <div><CumulativeReviews data={this.props.cumulativeReviews} /></div>
    } else if (this.state.type === 'two') {
      content = <div><a href='#'>AVATAR</a></div>
    } else if (this.state.type === 'three') {
      content = <div></div>
    }
    return (
      <div className='full-width full-height white-bg'>
        <Tabs defaultActiveKey={1} animation={false} id='noanim-tab-example'>
          <Tab eventKey={1} title='产品介绍'>
            <ProductIntroduction data={this.props.profile} />
          </Tab>
          <Tab eventKey={2} title='公司背景'>
            <CompanyProfile data={this.props.profile} />
          </Tab>
          <Tab eventKey={3} title='服务条款'><ServiceTerms data={this.props.profile} /></Tab>
          <Tab eventKey={4} title='累积评价'>
            <div className={styles.reviewbg}>
              <div className={classNames(styles.ratting_bg, styles.pdleft, 'form-group')}>
                <input onClick={this._onSelect1} type='radio' name='optionsRadios' />&nbsp;&nbsp;全部&nbsp;&nbsp;
                <input onClick={this._onSelect2} type='radio' name='optionsRadios' />&nbsp;&nbsp;追评（9）&nbsp;&nbsp;
                <input onClick={this._onSelect3} type='radio' name='optionsRadios' />&nbsp;&nbsp;图片（9）
              </div>
              {content}
            </div>
          </Tab>
        </Tabs>
      </div>
    )
  }
  componentDidMount () {
    this.props.actions.getCumulativeReviews()
    this.props.actions.getCompanyProfile()
  }
  _onSelect1 () {
    this.setState({type: 'one'})
  }
  _onSelect2 () {
    this.setState({type: 'two'})
  }
  _onSelect3 () {
    this.setState({type: 'three'})
  }
}
DetailedProductDescription.propTypes = {
  actions: React.PropTypes.object.isRequired,
  cumulativeReviews: React.PropTypes.object,
  profile: React.PropTypes.object
}
export default connect(mapStateToProps, mapDispatchToProps)(DetailedProductDescription)
