import Immutable, { Record, List } from 'immutable'
import apiFactory from 'api/apiFactory'
const api = apiFactory()
import companyData from './CompanyData'
/**
 * 获取公司简介信息
 * @type {string}
 */
export const GET_COMPANYSUMMARY_REQUEST = 'GET_COMPANYSUMMARY_REQUEST'
export const GET_COMPANYSUMMARY_SUCCESS = 'GET_COMPANYSUMMARY_SUCCESS'
export const GET_COMPANYSUMMARY_FAILURE = 'GET_COMPANYSUMMARY_FAILURE'
 
export function getCompanySummaryRequest () {
  return {type: GET_COMPANYSUMMARY_REQUEST}
}
export function getCompanySummarySuccess (json) {
  return {
    type: GET_COMPANYSUMMARY_SUCCESS,
    payload: json
  }
}
export function getCompanySummaryFailure (error) {
  return {
    type: GET_COMPANYSUMMARY_FAILURE,
    payload: error
  }
}
export function getCompanySummary () {
  return dispatch => {
    dispatch(getCompanySummaryRequest())
    return api.products.categories.getProductCategoryList('james')
    .then(json => {
      dispatch(getCompanySummarySuccess(json))
    }).catch(error => {
      dispatch(getCompanySummaryFailure(error))
    })
  }
}
export const actions = {
  getCompanySummary
}
let InitialState = new Record({
  isLaunching: false,
  companySummary: new List()
})
const initialState = new InitialState()
export default function companySummaryReducer (state = initialState, action) {
  if (!(state instanceof InitialState)) return initialState.merge(state)
  switch (action.type) {
    case GET_COMPANYSUMMARY_REQUEST:
      return state.set('isLaunching', false)
    case GET_COMPANYSUMMARY_SUCCESS:
      /* var companySummary = Immutable.fromJS(JSON.parse(action.payload)) */
      var companySummary = Immutable.fromJS(companyData)
      return state.set('isLaunching', false).set('companySummary', companySummary)
    case GET_COMPANYSUMMARY_FAILURE:
      return state.set('isLaunching', false)
  }
  return state
}
