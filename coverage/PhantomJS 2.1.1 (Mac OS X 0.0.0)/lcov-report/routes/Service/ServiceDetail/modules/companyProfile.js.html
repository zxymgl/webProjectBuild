import Immutable, { Record, List } from 'immutable'
import apiFactory from 'api/apiFactory'
const api = apiFactory()
import companyData from './CompanyData'
/**
 * 获取公司背景信息
 * @type {string}
 */
export const GET_BACKGROUND_REQUEST = 'GET_BACKGROUND_REQUEST'
export const GET_BACKGROUND_SUCCESS = 'GET_BACKGROUND_SUCCESS'
export const GET_BACKGROUND_FAILURE = 'GET_BACKGROUND_FAILURE'
export function getCompanyProfileRequest () {
  return {type: GET_BACKGROUND_REQUEST}
}
export function getCompanyProfileSuccess (json) {
  return {
    type: GET_BACKGROUND_SUCCESS,
    payload: json
  }
}
export function getCompanyProfileFailure (error) {
  return {
    type: GET_BACKGROUND_FAILURE,
    payload: error
  }
}
export function getCompanyProfile () {
  return dispatch => {
    dispatch(getCompanyProfileRequest())
    return api.products.categories.getProductCategoryList('CTX2000')
    .then(json => {
      dispatch(getCompanyProfileSuccess(json))
    }).catch(error => {
      dispatch(getCompanyProfileFailure(error))
    })
  }
}
export const actions = {
  getCompanyProfile
}
let InitialState = Record({
  isLaunching: false,
  profile: new List()
})
const initialState = new InitialState()
export default function backgroundReducer (state = initialState, action) {
  if (!(state instanceof InitialState)) return initialState.merge(state)
  switch (action.type) {
    case GET_BACKGROUND_REQUEST:
      return state.set('isLaunching', false)
    case GET_BACKGROUND_SUCCESS:
      /* var profile = Immutable.fromJS(JSON.parse(action.payload)) */
      var profile = Immutable.fromJS(companyData)
      return state.set('isLaunching', false).set('profile', profile)
    case GET_BACKGROUND_FAILURE:
      return state.set('isLaunching', false)
  }
  return state
}
