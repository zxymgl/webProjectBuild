import React from 'react'
import Pagination from 'react-bootstrap/lib/Pagination'
import Button from 'react-bootstrap/lib/Button'
/*
//import styles from './PaginationNavigationButtons.scss'
*/
class PaginationNavigationButtons extends React.Component {
  constructor (props) {
    super(props)
    this.state = {
      activePage: 1
    }
    this._bind('handleSelect', '_onClick')
  }
  _bind (...methods) {
    methods.forEach((method) => {
      this[method] = this[method].bind(this)
    })
  }
  render () {
    return (
      <div className='col-md-12'>
        <div className='col-md-1'></div>
        <div className='col-md-10'>
          <div className='col-md-12'>
            <div className='col-md-3'></div>
            <div className='col-md-6'>
              <span>
                <Pagination
                  ellipsis
                  boundaryLinks
                  items={45}
                  maxButtons={5}
                  activePage={this.state.activePage}
                  onSelect={this.handleSelect} />
              </span>
              <span>
                <Button bsSize='small' onClick={this._onClick}>下一页</Button>
              </span>
            </div>
            <div className='col-md-3'></div>
          </div>
        </div>
        <div className='col-md-1'></div>
      </div>
    )
  }
  _onClick () {
  }
  handleSelect (eventKey) {
    this.setState({
      activePage: eventKey
    })
  }
}
PaginationNavigationButtons.propTypes = {
  data: React.PropTypes.object
}
export default PaginationNavigationButtons
