import Immutable, { Record, List } from 'immutable'
import apiFactory from 'api/apiFactory'
const api = apiFactory()
import serviceListData from './ServiceListData'
import serviceListTypeData from './ServiceSubcategoryNavigationData'
/**
 * 获取服务商品列表
 * @type {string}
 */
export const GET_SERVICELIST_REQUEST = 'GET_SERVICELIST_REQUEST'
export const GET_SERVICELIST_SUCCESS = 'GET_SERVICELIST_SUCCESS'
export const GET_SERVICELIST_FAILURE = 'GET_SERVICELIST_FAILURE'
export function getServiceListRequest () {
  return {type: GET_SERVICELIST_REQUEST}
}
export function getServiceListSuccess (json) {
  return {
    type: GET_SERVICELIST_SUCCESS,
    payload: json
  }
}
export function getServiceListFailure (error) {
  return {
    type: GET_SERVICELIST_FAILURE,
    payload: error
  }
}
export function getServiceList () {
  return dispatch => {
    dispatch(getServiceListRequest())
    return api.products.categories.getProductCategoryList('DUMMY_ID')
    .then(json => {
      dispatch(getServiceListSuccess(json))
    }).catch(error => {
      dispatch(getServiceListFailure(error))
    })
  }
}
/**
 * 获取服务商品列表顶部分类检索信息
 * @type {string}
 */
export const GET_SERVICELISTTYPE_REQUEST = 'GET_SERVICELISTTYPE_REQUEST'
export const GET_SERVICELISTTYPE_SUCCESS = 'GET_SERVICELISTTYPE_SUCCESS'
export const GET_SERVICELISTTYPE_FAILURE = 'GET_SERVICELISTTYPE_FAILURE'

export function getServiceListTypeSuccess (json) {

export function getServiceListTypeFailure (error) {

export const actions = {
  getServiceList,
  getServiceListType
}
let InitialState = new Record({
  isLaunching: false,
  serviceListData: new List(),
  isLoading: false,
  serviceListTypeData: new List()
})
const initialState = new InitialState()
export default function serviceListReducer (state = initialState, action) {
  if (!(state instanceof InitialState)) return initialState.merge(state)
  switch (action.type) {
    case GET_SERVICELIST_REQUEST:
      return state.set('isLaunching', false)
    case GET_SERVICELIST_SUCCESS:
      var servicelist = Immutable.fromJS(serviceListData)
      return state.set('isLaunching', false).set('serviceListData', servicelist)
    case GET_SERVICELIST_FAILURE:
      return state.set('isLaunching', false)
    case GET_SERVICELISTTYPE_REQUEST:
      return state.set('isLoading', false)
    case GET_SERVICELISTTYPE_SUCCESS:
      var serviceListType = Immutable.fromJS(serviceListTypeData)
      return state.set('isLoading', false).set('serviceListTypeData', serviceListType)
    case GET_SERVICELISTTYPE_FAILURE:
      return state.set('isLoading', false)
  }
  return state
}
