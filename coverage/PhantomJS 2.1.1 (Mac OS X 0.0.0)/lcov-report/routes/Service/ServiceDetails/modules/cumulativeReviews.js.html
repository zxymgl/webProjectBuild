import Immutable, { Record, List } from 'immutable'
import apiFactory from 'api/apiFactory'
const api = apiFactory()
import reviewsData from './ReviewsData'
/**
 * 获取服务累积评价信息
 * @type {string}
 */
export const GET_REVIEWLIST_REQUEST = 'GET_REVIEWLIST_REQUEST'
export const GET_REVIEWLIST_SUCCESS = 'GET_REVIEWLIST_SUCCESS'
export const GET_REVIEWLIST_FAILURE = 'GET_REVIEWLIST_FAILURE'
export function getCumulativeReviewsRequest () {
  return {type: GET_REVIEWLIST_REQUEST}
}
export function getCumulativeReviewsSuccess (json) {
  return {
    type: GET_REVIEWLIST_SUCCESS,
    payload: json
  }
}
export function getCumulativeReviewsFailure (error) {
  return {
    type: GET_REVIEWLIST_FAILURE,
    payload: error
  }
}
export function getCumulativeReviews () {
  return dispatch => {
    dispatch(getCumulativeReviewsRequest())
    return api.products.categories.getProductCategoryList('DUMMY_ID')
    .then(json => {
      dispatch(getCumulativeReviewsSuccess(json))
    }).catch(error => {
      dispatch(getCumulativeReviewsFailure(error))
    })
  }
}
export const actions = {
  getCumulativeReviews
}
let InitialState = Record({
  isLoading: false,
  cumulativeReviewsList: new List()
})
const initialState = new InitialState()
export default function serviceReviewListReducer (state = initialState, action) {
  if (!(state instanceof InitialState)) return initialState.merge(state)
  switch (action.type) {
    case GET_REVIEWLIST_REQUEST:
      return state.set('isLoading', false)
    case GET_REVIEWLIST_SUCCESS:
      /* var reviewlist = Immutable.fromJS(JSON.parse(action.payload)) */
      var reviewlist = Immutable.fromJS(reviewsData)
      return state.set('isLoading', false).set('cumulativeReviewsList', reviewlist)
    case GET_REVIEWLIST_FAILURE:
      return state.set('isLoading', false)
  }
  return state
}
