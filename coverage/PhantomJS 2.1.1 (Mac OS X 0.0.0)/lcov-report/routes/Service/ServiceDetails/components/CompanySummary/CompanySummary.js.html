import React from 'react'
class CompanySummary extends React.Component {
  render () {
    return (
      <div>
        {this.props.data.map(function (cp, index) {
          return (
            <div key={index}>
              <h3>{cp.get('companyName')}</h3>
              <br />
              <h4>{cp.get('summary')}</h4>
              <br />
              <h3>{cp.get('type')}</h3>
            </div>
          )
        })}
      </div>
    )
  }
}
CompanySummary.propTypes = {
  data: React.PropTypes.object
}
export default CompanySummary
