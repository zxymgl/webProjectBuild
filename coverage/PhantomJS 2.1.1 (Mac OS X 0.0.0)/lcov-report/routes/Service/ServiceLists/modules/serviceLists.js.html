import Immutable, { Record, List } from 'immutable'
import apiFactory from 'api/apiFactory'
const api = apiFactory()
import servicelists from './ServiceListsData'
 
/**
 * 获取服务商品列表
 * @type {string}
 */
export const GET_SERVICELIST_REQUEST = 'GET_SERVICELIST_REQUEST'
export const GET_SERVICELIST_SUCCESS = 'GET_SERVICELIST_SUCCESS'
export const GET_SERVICELIST_FAILURE = 'GET_SERVICELIST_FAILURE'
 
export function getServiceListsRequest () {
  return {type: GET_SERVICELIST_REQUEST}
}
export function getServiceListsSuccess (json) {
  return {
    type: GET_SERVICELIST_SUCCESS,
    payload: json
  }
}
export function getServiceListsFailure (error) {
  return {
    type: GET_SERVICELIST_FAILURE,
    payload: error
  }
}
export function getServiceLists () {
  return dispatch => {
    dispatch(getServiceListsRequest())
    return api.products.categories.getProductCategoryList('CJZC3PL')
    .then(json => {
      dispatch(getServiceListsSuccess(json))
    }).catch(error => {
      dispatch(getServiceListsFailure(error))
    })
  }
}
 
export const actions = {
  getServiceLists
}
let InitialState = new Record({
  isLaunching: false,
  serviceLists: new List()
})
 
var initialState = new InitialState()
export default function serviceListReducer (state = initialState, action) {
  if (!(state instanceof InitialState)) return initialState.merge(state)
  switch (action.type) {
    case GET_SERVICELIST_REQUEST:
      return state.set('isLaunching', false)
    case GET_SERVICELIST_SUCCESS:
      /* var lists = Immutable.fromJS(JSON.parse(action.payload)) */
      var lists = Immutable.fromJS(servicelists)
      return state.set('isLaunching', false).set('serviceLists', lists)
    case GET_SERVICELIST_FAILURE:
      return state.set('isLaunching', false)
  }
  return state
}
