import React from 'react'
import classNames from 'classnames'
import styles from './ProvinceAndCitySelection.scss'
import stepsStyles from 'styles/inspinia/plugins/steps/jquery.steps.scss'
 
var provinceId = 0
let cities = []
class ProvinceAndCitySelection extends React.Component {
  constructor (props) {
    super(props)
    this.state = {
      currentType: 0
    }
    provinceId = 0
    this._bind('_onChangeOption', '_onSelectOption')
  }
  _bind (...methods) {
    methods.forEach((method) => {
      this[method] = this[method].bind(this)
    })
  }
  render () {
    return (
      <div>
        <div>
          <div className={classNames('clearfix', stepsStyles.steps)}>
            <div className='col-md-3'>
              <span>选择省份:</span>
              &#160;&#160;&#160;
              <select className={classNames(styles.selectoor)} id='ctx2000' onChange={this._onChangeOption}>
                {this.props.data.map((v, i) => {
                  return (<option id={i} key={i} className={styles.selectorr} value={i}>
                    {v.get('provinceName')}
                  </option>)
                })}
              </select>
            </div>
            <div className='col-md-3'>
              <select className={styles.selectoor} id='ctx3000' onChange={this._onSelectOption}>
                {this.props.data.map((v, i) => {
                  this.props.data.get(provinceId).get('cityList').map((vv, ii) => {
                    return cities.push(<option key={ii} className={styles.selectorr} value={ii}>{vv.get('cityName')}</option>)
                  })
                })}
                {cities}
              </select>
              &#160;&#160;&#160;&#160;&#160;&#160;
              <button type='submit' className={styles.pal}>
                <strong>确定</strong>
              </button>
            </div>
            <div className='col-md-4'>
              <span>直接搜索:</span>
              &#160;&#160;&#160;
              <input className={styles.selectoor} type='text'
                     defaultValue='请输入城市名称' onChange={this._onChange} />
              &#160;&#160;&#160;&#160;&#160;&#160;
              <button type='submit' className={styles.pal}>
                <strong>搜索</strong>
              </button>
            </div>
            <div className='col-md-2'></div>
          </div>
        </div>
      </div>
    )
  }
  componentDidUpdate (prevProps, prevState) {
  }
  _onChangeOption () {
    var obj = window.document.getElementById('ctx2000')
    provinceId = obj.options[obj.selectedIndex].id
    this.setState({currentType: provinceId})
    cities.splice(0, cities.length)
  }
  _onSelectOption () {
  }
  _onChange () {
  }
}
ProvinceAndCitySelection.propTypes = {
  data: React.PropTypes.object
}
export default ProvinceAndCitySelection
