import React from 'react'
import classNames from 'classnames'
import styles from './HotCities.scss'
import stepsStyles from '../../../../styles/inspinia/plugins/steps/jquery.steps.scss'
 
class HotCities extends React.Component {
  render () {
    return (
      <div className='row'>
        <div className={classNames(styles.line, 'ibox-content')}>
          <div className={classNames('clearfix', stepsStyles.steps)}>
            <ul>
              <li className={classNames(styles.lii, styles.aa)}>
                <h3 className={styles.font}><strong>热门城市</strong></h3>
              </li>
              {this.props.data.map((v, i) => {
                return <li key={i} className={classNames(styles.lii, styles.aa)}>
                  <a className={styles.ac}><h3>{v.get('cityName')}</h3></a>
                </li>
              })}
            </ul>
          </div>
        </div>
      </div>
    )
  }
}
HotCities.propTypes = {
  data: React.PropTypes.object
}
export default HotCities
