import React from 'react'
import classNames from 'classnames'
import styles from './AlphabetArea.scss'
import ReactMenuAim from 'components/ReactMenuAim'
class AlphabetArea extends ReactMenuAim {
  render () {
    return (
      <div>
        <div>
          {this.props.data.map((v, i) => {
            return (
              <div key={i}>
                <div className={styles.left}>
                  <ul>
                    <li className={classNames(styles.lii, styles.l)}>
                      <a className={styles.a}><h3>{v.get('alpha')}</h3></a>
                    </li>
                    {v.get('cityList').map((alpha, ii) => {
                      return (
                        <li key={ii} className={classNames(styles.lii, styles.l)}>
                          <a className={styles.aa}><h3>{alpha.get('cityName')}</h3></a>
                        </li>
                      )
                    })}
                  </ul>
                </div>
                <br />
                <br />
              </div>
            )
          })}
        </div>
      </div>
    )
  }
}
AlphabetArea.propTypes = {
  data: React.PropTypes.object
}
export default AlphabetArea
