export function getSelectionRequest () {
  return {type: GET_SELECTIONDATA_REQUEST}
}
export function getSelectionSuccess (json) {
  return {
    type: GET_SELECTIONDATA_SUCCESS,
    payload: json
  }

export function getSelectionFailure (error) {
  return {
    type: GET_SELECTIONDATA_FAILURE,
    payload: error
  }

export function getSelection () {
  return dispatch => {
    dispatch(getSelectionRequest())
    return api.products.categories.get('')
      .then(json => {
        dispatch(getSelectionSuccess(json))
      }).catch(error => {
        dispatch(getSelectionFailure(error))
      })
  }
}

export const actions = {
  getSelection,
  getHotCityList,
  getCityByAlpha,
  getProvincesAndCities
}
let InitialState = new Record({
  isLauncher: false,
  selection: new List(),
  isLoading: false,
  hotCityList: new List(),
  isFetching: false,
  cityAlpha: new List(),
  isRequesting: false,
  provincesAndCitiesList: new List()
})
const initialState = new InitialState()
export default function SelectionReducer (state = initialState, action) {
  if (!(state instanceof InitialState)) return initialState.merge(state)
  switch (action.type) {
    case GET_SELECTIONDATA_REQUEST:
      return state.set('isLauncher', false)
    case GET_SELECTIONDATA_SUCCESS:
      var selectData = Immutable.fromJS(selectionData)
      return state.set('isLauncher', false).set('selection', selectData)
    case GET_SELECTIONDATA_FAILURE:
      return state.set('isLauncher', false)
    case GET_HOTCITY_REQUEST:
      return state.set('isLoading', false)
    case GET_HOTCITY_SUCCESS:
      var hotCityList = Immutable.fromJS(hotCityData)
      return state.set('isLoading', false).set('hotCityList', hotCityList)
    case GET_HOTCITY_FAILURE:
      return state.set('isLoading', false)
    case GET_SELECTCITY_REQUEST:
      return state.set('isFetching', false)
    case GET_SELECTCITY_SUCCESS:
      var cities = Immutable.fromJS(selectionData)
      return state.set('isFetching', false).set('cityAlpha', cities)
    case GET_SELECTCITY_FAILURE:
      return state.set('isFetching', false)
    case GET_PROVINCESANDCITIES_REQUEST:
      return state.set('isRequesting', false)
    case GET_PROVINCESANDCITIES_SUCCESS:
      var provincesAndCities = Immutable.fromJS(provincesAndCitiesData)
      return state.set('isRequesting', false).set('provincesAndCitiesList', provincesAndCities)
    case GET_PROVINCESANDCITIES_FAILURE:
      return state.set('isRequesting', false)
  }
  return state
}
