import React, { PropTypes } from 'react'
import curStyle from './Cascader.scss'
import classNames from 'classnames'
class Menu extends React.Component {
  constructor (props) {
    super(props)
    this.state = {
    }
    this._onClick = this._onClick.bind(this)
  }
  static propTypes = {
    onItemClick: PropTypes.func,
    label: PropTypes.string,
    reactKey: PropTypes.number,
    activeLevel: PropTypes.number,
    categoryId: PropTypes.string,
    isActive: PropTypes.bool
  }
 
  render () {
    let curClassName
    if (this.props.isActive) {
      curClassName = classNames(curStyle.active, 'm-t-sm', curStyle.noborderRadius)
    } else {
      curClassName = classNames('m-t-sm',curStyle.noborderRadius)
    }
    return (
      <li className={curClassName} onClick={this._onClick}>
        <label className='m-l'>{this.props.label}</label> <span className='glyphicon glyphicon-menu-right pull-right '></span>
      </li>
    )
  }
  _onClick () {
    this.props.onItemClick(this.props.activeLevel, this.props.reactKey ,this.props.categoryId)
  }
}
 
export default Menu
