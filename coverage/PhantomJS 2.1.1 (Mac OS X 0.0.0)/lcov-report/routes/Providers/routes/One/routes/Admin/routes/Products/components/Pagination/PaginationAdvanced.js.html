import React, { PropTypes } from 'react'
import { Pagination } from 'react-bootstrap'
 
class GlidingLineNavigationBar extends React.Component {
  constructor (props) {
    super(props)
    this.state = {
      activePage: 1
    }
    this._handleSelect = this._handleSelect.bind(this)
  }
  static propTypes = {
    callBackParent: PropTypes.func
  }
  _handleSelect =(eventKey) => {
    this.setState({
      activePage: eventKey
    })
  }
  render () {
    return (
      <Pagination
        prev
        next
        first
        last
        ellipsis
        boundaryLinks
        items={20}
        maxButtons={5}
        activePage={this.state.activePage}
        onSelect={this.handleSelect} />
    )
  }
}
 
export default GlidingLineNavigationBar
