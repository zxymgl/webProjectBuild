import React, { PropTypes } from 'react'
import classNames from 'classnames'
import { Map, List } from 'immutable'
import ImmutablePropTypes from 'react-immutable-proptypes'
import Menu from './Menu'
class Cascader extends React.Component {
  constructor (props) {
    super(props)
    this.state = {
      activeLevel: 0,
      nodeIndex: [-1, -1, -1, -1, -1],
    }
  }
  static propTypes = {
    options: PropTypes.array
  }
  _onClick = (level, index ,categoryId) => {
    let newNodeIndex = this.state.nodeIndex
    newNodeIndex.splice(level, 1, index)
    newNodeIndex.map((item, key) => {
      if (key > level) {
        newNodeIndex.splice(key, 1, -1)
      }
    })
    let nextLevel = level + 1
    this.setState({
      activeLevel: nextLevel,
      nodeIndex: newNodeIndex
    })
   this.props.callBack(categoryId)
  }
  render () {
    // console.log(this.props.options)
    // console.log(JSON.stringify(this.props.options))
    /* 第一层数据 this.props.options */
    /* 第二层数据 this.props.options. */
    let nodeIndex1 = this.state.nodeIndex[0]
    let node1object = this.props.options.get(nodeIndex1)
    let arr1
    if (!node1object) {
      arr1 = new List()
    } else {
      arr1 = node1object.get('children')
      if (!arr1)arr1 = new List()
    }
    /* 第三层数据 this.props.options */
    let nodeIndex2 = this.state.nodeIndex[1]
    let node2object
    let arr2
    if (nodeIndex2 >= 0) {
      node2object = arr1.get(nodeIndex2)
      if (!node2object) {
        arr2 = new List()
      } else {
        arr2 = node2object.get('children')
        if (!arr2)arr2 = new List()
      }
    }
    /* 第四层数据 this.props.options */
    let nodeIndex3 = this.state.nodeIndex[2]
    let node3object
    let arr3
    if (nodeIndex3 >= 0) {
      node3object = arr2.get(nodeIndex3)
      if (!node3object) {
        arr3 = new List()
      } else {
        arr3 = node3object.get('children')
        if (!arr3)arr3 = new List()
      }
    }
    /* 根据activeLevel 调整css */
    var subMenuStyle = {
      padding: 0
    }
    return (
      <div className='col-md-8 table-bordered '>
        <div className='col-md-4 m-t-sm ' style={subMenuStyle}>
          <ul className='list-unstyled' >
            {this.props.options.map(function (menu, index) {
              let nodeIndex1 = this.state.nodeIndex[0]
              let isActive
              if (nodeIndex1 === index) {
                isActive = true
              } else {
                isActive = false
              }
              return (
                <Menu key={index} reactKey={index} activeLevel={0} label={menu.get('productCategory').get('categoryName')} categoryId={menu.get('productCategory').get('productCategoryId')} onItemClick={this._onClick} isActive={isActive} />
              )
            }.bind(this))}
          </ul>
        </div>
        <div className='col-md-4 m-t-sm' style={subMenuStyle}>
          {this.state.activeLevel >= 1 && arr1
            ? (
            <ul className={classNames('list-unstyled')} >
              {arr1.map(function (subMenu, index) {
                let nodeIndex1 = this.state.nodeIndex[1]
                let isActive
                if (nodeIndex1 === index) {
                  isActive = true
                } else {
                  isActive = false
                }
                return (
                  <Menu key={index} reactKey={index} activeLevel={1} label={subMenu.get('productCategory').get('categoryName')} categoryId={subMenu.get('productCategory').get('productCategoryId')} onItemClick={this._onClick} isActive={isActive} />
                )
              }.bind(this))}
            </ul>)
            : null
          }
        </div>
        <div className='col-md-4 m-t-sm' style={subMenuStyle} >
          {this.state.activeLevel >= 2 && arr2
            ? (
            <ul className='list-unstyled'>
              {arr2.map(function (subMenu, index) {
                let nodeIndex1 = this.state.nodeIndex[2]
                let isActive
                if (nodeIndex1 === index) {
                  isActive = true
                } else {
                  isActive = false
                }
                return (
                  <Menu key={index} reactKey={index} activeLevel={2} label={subMenu.get('productCategory').get('categoryName')} categoryId={subMenu.get('productCategory').get('productCategoryId')} onItemClick={this._onClick} isActive={isActive} />
                )
              }.bind(this))}
            </ul>)
            : null
          }
        </div>
        <div className='col-md-4' style={subMenuStyle}>
          {this.state.activeLevel >= 3 && arr3
          ? (
            <ul className='list-unstyled' >
              {arr3.map(function (subMenu, index) {
                let nodeIndex1 = this.state.nodeIndex[3]
                let isActive
                if (nodeIndex1 === index) {
                  isActive = true
                } else {
                  isActive = false
                }
                return (
                  <Menu key={index} reactKey={index} activeLevel={3} label={subMenu.get('productCategory').get('categoryName')} categoryId={subMenu.get('productCategory').get('productCategoryId')} onItemClick={this._onClick} isActive={isActive} />
                )
              }.bind(this))}
            </ul>)
          : null
        }
        </div>
      </div>
    )
  }
}
Cascader.defaultProps = {
  options: ImmutablePropTypes.list
}
Cascader.propTypes = {
  options: ImmutablePropTypes.list,
  callBack: PropTypes.func
}
export default Cascader
