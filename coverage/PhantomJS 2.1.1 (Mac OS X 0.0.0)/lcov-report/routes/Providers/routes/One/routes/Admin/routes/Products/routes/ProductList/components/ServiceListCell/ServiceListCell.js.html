import React, { PropTypes } from 'react'
import ImmutablePropTypes from 'react-immutable-proptypes'
 /* import assets */
import defaultImage from './assets/default.png'
/* import css */
import curStyle from './ServiceListCell.scss'
import config from '../../../../../../../../../../../../api/config'
class ServiceListCell extends React.Component {
  constructor (props) {
    super(props)
    this.state = {
      SwitchMenuIndex: 0
    }
  }
 
  render () {
 
    console.log(JSON.stringify(this.props.data))
    let headerImageId
    for (let i = 0; i < this.props.data.productContentList.size; i++) {
      let productContent = this.props.data.productContentList.get(i)
      if (productContent.productContentTypeEnumId === 'PcntImageUrlSmall') {
        headerImageId = productContent.productContentId
        break
      }
    }
    let headerImage
    if (headerImageId) {
      headerImage = config.baseUrl + '/s1/zc/products/contents/' + headerImageId + '/file'
      console.log(headerImage)
    } else {
      headerImage = defaultImage
    }
    return (
      <tr className='table-bordered '>
        <td >
          <label className={curStyle.seletTop}> <input type='checkbox' />&nbsp;&nbsp;</label>
          <div className={curStyle.imageLoac}>
            <img src={headerImage} height='100' width='150' />
          </div>
        </td>
        <td >
          <div className={curStyle.lable}>
            <p>{this.props.data.productName}</p>
          </div>
        </td>
        <td>
          <div className={curStyle.lable}>
            <p>{`¥${this.props.data.productPriceList.get(0) ? this.props.data.productPriceList.get(0).price : ''}`}</p>
            {this.props.type === 'temperature' ? <button className='btn-info no-borders'>修改服务</button> : null}
          </div>
        </td>
 
        <td>
          <div className={curStyle.lable}>
            <p>2016-05-1718:24:18</p>
            {this.props.type === 'temperature' ? <button className='btn-danger no-borders'>上架</button> : null}
            {this.props.type === 'onSale' ? <button className='btn-danger no-borders'>下架</button> : null}
          </div>
        </td>
      </tr>
    )
  }
}
ServiceListCell.propTypes = {
  callBackParent: PropTypes.func,
  type: PropTypes.string,
  data: ImmutablePropTypes.record
}
export default ServiceListCell
