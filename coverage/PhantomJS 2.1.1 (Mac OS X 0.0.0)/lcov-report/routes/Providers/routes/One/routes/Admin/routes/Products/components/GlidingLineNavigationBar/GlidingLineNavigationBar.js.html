import React, { PropTypes } from 'react'
 
import currentStyle from './GlidingLineNavigationBar.scss'
import { IndexLink, Link } from 'react-router'
 
class GlidingLineNavigationBar extends React.Component {
  constructor (props) {
    super(props)
    this.state = {
    }
  }
  static propTypes = {
 
  }
  render () {
    const pages = [
      [ 'list', '服务列表' ],
      [ 'new', '创建服务' ]
    ]
    let { partyId } = this.props.params
 
    return (
      <div>
        <div className='col-lg-12 border-bottom'>
          <ul className='nav navbar-nav '>
            {pages.map((page, index) => (
              <li key={page[0]} className={this.props.switchMenuIndex === index && currentStyle.active} >
                <Link to={`/providers/${partyId}/admin/products/${page[0]}`}>{page[1]}</Link>
              </li>
            ))}
          </ul>
        </div>
      </div>
    )
  }
}
GlidingLineNavigationBar.propTypes = {
  switchMenuIndex: PropTypes.number,
  params: PropTypes.any
}
export default GlidingLineNavigationBar
