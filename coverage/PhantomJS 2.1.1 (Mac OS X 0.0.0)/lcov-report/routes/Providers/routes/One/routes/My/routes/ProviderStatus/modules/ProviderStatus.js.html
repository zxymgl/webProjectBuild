export const getDashboardData = (productStoreId, query0, query1, query2, query3 , query4, query5, pagination) => dispatch => {
  dispatch({ type: GET_STORES_ONE_ORDERS_ALL_REQUEST })
  let promises = []
  promises.push(api.stores.one.orders.get(productStoreId, query0, pagination).then(json => {
    dispatch({ type: GET_STORES_ONE_ORDERS_ORDEROPEN_SUCCESS, payload: json })
  }))
 
  promises.push(api.stores.one.orders.get(productStoreId, query1, pagination).then(json => {
    dispatch({ type: GET_STORES_ONE_ORDERS_ORDERPLACED_SUCCESS, payload: json })
  }))
  promises.push(api.stores.one.orders.get(productStoreId, query2, pagination).then(json => {
    dispatch({ type: GET_STORES_ONE_ORDERS_ORDERPROCESSING_SUCCESS, payload: json })
  }))
  promises.push(api.stores.one.orders.get(productStoreId, query3, pagination).then(json => {
    dispatch({ type: GET_STORES_ONE_ORDERS_ORDERRESULTSUBMITTE_SUCCESS, payload: json })
  }))
  promises.push(api.stores.one.orders.get(productStoreId, query4, pagination).then(json => {
    dispatch({ type: GET_STORES_ONE_ORDERS_ORDERRESULTCONFIRMED_SUCCESS, payload: json })
  }))
  promises.push(api.stores.one.orders.get(productStoreId, query5, pagination).then(json => {
    dispatch({ type: GET_STORES_ONE_ORDERS_ORDERCOMPLETE_SUCCESS, payload: json })
  }))
  Promise.all(promises).then(values => {
    dispatch({ type: GET_STORES_ONE_ORDERS_ALL_SUCCESS })
  }).catch(error => {
    dispatch({ type: GET_STORES_ONE_ORDERS__ALL_FAILURE, payload: error })
  })
}
