        dispatch(getPartiesPersons_One_Failure(error))
      })
  }
}

export const getProviderInfo = partyId => dispatch => {
  dispatch({ type: GET_PROVIDER_INFO_ALL_REQUEST })
 
  let promises = []
  promises.push(api.parties.organizations.one.get(partyId).then(json => {
    dispatch({ type: GET_PROVIDER_INFO_SUCCESS, payload: json })
  }))
 
  promises.push(api.parties.persons.one.get(partyId).then(json => {
    dispatch({ type: GET_PERSON_INFO_SUCCESS, payload: json })
  }))
 
  Promise.all(promises).then(values => {
    dispatch({ type: GET_PROVIDER_INFO_ALL_SUCCESS })
  }).catch(error => {
    dispatch({ type: GET_PROVIDER_INFO_ALL_FAILURE, payload: error })
  })
}

const initialState = new InitialState()
 
export default function providerInfoReducer (state = initialState, action) {
  if (!(state instanceof InitialState)) return initialState.merge(state)
 
  let json
  switch (action.type) {
    // GET_PARTIES_PERSONS_ONE
    case GET_PARTIES_PERSONS_ONE_REQUEST:
      return state.set('isFetching', true)
 
    case GET_PARTIES_PERSONS_ONE_SUCCESS:
      if (!Immutable.fromJS(action.payload)) return null
      let pcMap = Immutable.fromJS(action.payload)
      return state.set('isFetching', false)
        .set('personInfo', pcMap.get('person'))
 
    case GET_PARTIES_PERSONS_ONE_FAILURE:
      return state.set('isFetching', false)
    // GET_PROVIDER_INFO
    case GET_PROVIDER_INFO_ALL_REQUEST:
      return state.set('isFetching', true)
    case GET_PROVIDER_INFO_ALL_SUCCESS:
      return state.set('isFetching', false)
    case GET_PROVIDER_INFO_ALL_FAILURE:
      return state.set('isFetching', false)
 
    case GET_PROVIDER_INFO_SUCCESS:
      json = action.payload
 
      return state.setIn(['provider', 'partyId'], json.partyId)
        .setIn(['provider', 'organizationName'], json.organizationName)
        .setIn(['provider', 'numEmployees'], json.numEmployees)
 
    case GET_PERSON_INFO_SUCCESS:
      json = JSON.parse(action.payload)
      return state.setIn('person', 'partyId', json.partyId)
        .setIn('person', 'firstName', json.firstName)
        .setIn('person', 'lastName', json.lastName)
        .setIn('person', 'personalTitle', json.personalTitle)
        .setIn('person', 'nickName', json.nickName)
        .setIn('person', 'gender', json.gender)
        .setIn('person', 'birthDate', json.birthDate)
  }
  return state
}
