}
 
export function getStoresOneOrders (productStoreId, query, pagination) {
  return dispatch => {
    dispatch(getStoresOneOrdersRequest())
    return api.stores.one.orders.get(productStoreId, query, pagination)
      .then(json => {
        dispatch(getStoresOneOrdersSuccess(json, query.statusId))
      }).catch(error => {
        dispatch(getStoresOneOrdersFailure(error))
      })
  }
}
