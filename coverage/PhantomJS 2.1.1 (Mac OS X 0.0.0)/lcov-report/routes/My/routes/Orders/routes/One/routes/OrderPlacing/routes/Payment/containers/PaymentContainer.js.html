import React from 'react'
 
import OrderSuccess from '../components/OrderSuccess/OrderSuccess'
import PaymentCateWays from '../components/PaymentMethods/PaymentMethods'
import PaymentStatus from '../components/PayStatus/PaymentStatus'
import OrderAuctionNavView from '../../../components/OrderAuctionNav'
 
class PaymentContainer extends React.Component {
  constructor (props) {
    super(props)
    this.state = {
      paymentStatus: false
    }
  }
  _onChildChanged = (newState) => {
    this.setState({
      paymentStatus: newState
    })
  }
  _renderView = () => {
    switch (this.state.paymentStatus) {
      case false:
        return (
          <div className='col-lg-12'>
            <OrderAuctionNavView  switchMenuIndex={1}/>
            <OrderSuccess>.</OrderSuccess>
            <PaymentCateWays callBackPaymentStatus={this._onChildChanged}>.</PaymentCateWays>
          </div>
        )
      case true:
        return (
          <PaymentStatus>.</PaymentStatus>
        )
    }
  }
  render () {
    return (
     this._renderView()
    )
  }
}
export default PaymentContainer
