import React, { PropTypes } from 'react'
 
import YStar from '../assets/yStar.png'
import GStar from '../assets/gStar.png'
class StarRating extends React.Component {
  constructor (props) {
    super(props)
    this.state = {
    }
    this._imgClick = this._imgClick.bind(this)
  }
  _imgClick =() => {
    if (this.props.onItemClick) {
      this.props.onItemClick(this.props.startLeavel, this.props.starKind)
    }
  }
  static propTypes = {
    onItemClick: PropTypes.func,
    isLight: PropTypes.bool,
    starKind: PropTypes.string,
    startLeavel: PropTypes.number
  }
  render () {
    if (!this.props.isLight) {
      return (
        <img src={GStar} onClick={this._imgClick} />
      )
    } else {
      return (
        <img src={YStar} onClick={this._imgClick} />
      )
    }
  }
}
export default StarRating
