import React from 'react'
import ScoreStatic from './ScoreData'
 
import Immutable from 'immutable'
import StarRating from '../components/StarRating'
class Score extends React.Component {
 
  constructor (props) {
    super(props)
    var pcList = Immutable.fromJS(ScoreStatic)
    this.state = {
      staticData: pcList,
      descriptionLevel: 5,
      serviceAttitudeLevel: 5,
      logisticsLevel: 5
    }
    this._renderStart = this._renderStart.bind(this)
    this._changeLevel = this._changeLevel.bind(this)
  }
  _changeLevel = (level, kind) => {
    switch (kind) {
      case 'description':
        this.setState({
          descriptionLevel: level
        })
        break
      case 'serviceAttitude':
        this.setState({
          serviceAttitudeLevel: level
        })
        break
      case 'logistics':
        this.setState({
          logisticsLevel: level
        })
        break
    }
  }
  _renderStart = (startCount, cstarKind) => {
    var doms = []
    for (let i = 0; i < startCount; i++) {
      doms.push(<StarRating key={i + 1} isLight={true} starKind={cstarKind} startLeavel={i + 1} onItemClick={this._changeLevel}/>)
    }
    for (let i = 0; i < (5 - startCount); i++) {
      doms.push(<StarRating key={startCount + i + 1} isLight={false} starKind={cstarKind} startLeavel={startCount + i + 1} onItemClick={this._changeLevel} />)
    }
    return doms
  }
  render () {
    return (
      <div className='col-sm-7 col-xs-offset-5 icons-box h3'>
        <ul className='todo-list m-t ui-sortable'>
          <li>
            <span><small>{this.state.staticData.get(0).get('contentind')}</small></span>
            {this._renderStart(this.state.descriptionLevel, 'description')}
            <span className='text-danger'>&nbsp;{this.state.descriptionLevel}分&nbsp;</span>
            <span>{this.state.staticData.get(0)
              .get('descriptionList').get(this.state.descriptionLevel - 1).get('description')}</span>
          </li>
          <li>
            <span><small>{this.state.staticData.get(1)
              .get('contentind')}</small></span>
            {this._renderStart(this.state.serviceAttitudeLevel, 'serviceAttitude')}
            <span className='text-danger'>&nbsp;{this.state.serviceAttitudeLevel}分&nbsp;</span>
            <span>{this.state.staticData.get(0)
              .get('descriptionList').get(this.state.serviceAttitudeLevel - 1).get('description')}</span>
          </li>
          <li>
            <span><small>{this.state.staticData.get(2)
              .get('contentind')}</small></span>
            {this._renderStart(this.state.logisticsLevel, 'logistics')}
            <span className='text-danger'>&nbsp;{this.state.logisticsLevel}分&nbsp;</span>
            <span>{this.state.staticData.get(0)
              .get('descriptionList').get(this.state.logisticsLevel - 1).get('description')}</span>
          </li>
        </ul>
      </div>
    )
  }
}
export default Score
