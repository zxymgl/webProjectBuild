import { combineReducers } from 'redux'
import orderListReducer from './modules/myOrderList'
 
export const reducer = combineReducers({
  orderList: orderListReducer
})
 
export default (store) => {
  return {
    path: 'list',
    getComponent (nextState, cb) {
      require.ensure([], (require) => {
        const OrderListContainer = require('./containers/OrderListContainer').default
        cb(null, OrderListContainer)
      }, 'my')
    },
    
    getChildRoutes (location, cb) {
      require.ensure([], (require) => {
        const AllOrdersRoute = require('./routes/AllOrders').default
        const OrderOpen = require('./routes/OrderOpen').default
        const OrderPlaced = require('./routes/OrderPlaced').default
        const OrderResultSubmitted = require('./routes/OrderResultSubmitted').default
        const ToBeReviewed = require('./routes/ToBeReviewed').default
        cb(null, [
          AllOrdersRoute(store),
          OrderOpen(store),
          OrderPlaced(store),
          OrderResultSubmitted(store),
          ToBeReviewed(store)
        ])
      })
    }
  }
}
