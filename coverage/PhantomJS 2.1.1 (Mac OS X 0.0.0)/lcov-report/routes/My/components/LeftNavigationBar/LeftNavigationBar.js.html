import React from 'react'
import { Link } from 'react-router'
import styles from './LeftNavigationBar.scss'
import Orders from '../../routes/Orders'
import Reviews from '../../routes/Reviews'
import Messages from '../../routes/Messages'
 
class LeftNavigationBar extends React.Component {
  constructor (props) {
    super(props)
    this.state = {
      current: 'one'
    }
    this._bind('_onMessages', '_onOrders', '_onReviews', '_renderViewer')
  }
  _bind (...methods) {
    methods.forEach((method) => {
      this[method] = this[method].bind(this)
    })
  }
  _renderViewer = () => {
    switch (this.state.current) {
      case 'one':
        return <Orders />
      case 'two':
        return <Reviews />
      case 'three':
        return <Messages />
    }
  }
  render () {
    return (
      <nav className='navbar-default navbar-static-side' role='navigation'>
        <div className='sidebar-collapse'>
          <ul className={styles.navi} id='side-menu'>
            <li >
              <Link className={styles.lll} to='/my/orders/list/all'>
                <h4>订单中心</h4>
              </Link>
            </li>
            <li>
              <Link to='/my/orders/list/all'>
                <h4>全部订单</h4>
              </Link>
            </li>
            <li>
              <Link to='/my/reviews'>
                <h4>我的评价</h4>
              </Link>
            </li>
            <li>
              <Link to='/my/messages'>
                <h4>我的消息</h4>
              </Link>
            </li>
          </ul>
        </div>
      </nav>
    )
  }
  _onOrders () {
    this.setState({current: 'one'})
  }
  _onMessages () {
    this.setState({current: 'two'})
  }
  _onReviews () {
    this.setState({current: 'three'})
  }
}
 
LeftNavigationBar.propTypes = {
  data: React.PropTypes.object,
  status: React.PropTypes.object
}
 
export default LeftNavigationBar
