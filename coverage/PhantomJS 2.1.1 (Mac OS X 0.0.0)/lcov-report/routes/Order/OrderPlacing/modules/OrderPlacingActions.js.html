export const POST_HOME_ORDERS_ADDONE_REQUEST = 'POST_HOME_ORDERS_ADDONE_REQUEST'
export const POST_HOME_ORDERS_ADDONE_SUCCESS = 'POST_HOME_ORDERS_ADDONE_SUCCESS'
export const POST_HOME_ORDERS__ADDONE_FAILURE = 'POST_HOME_ORDERS__ADDONE_FAILURE'
 
export function postHomeOrdersAddOneRequest () {
  return { type: POST_HOME_ORDERS_ADDONE_REQUEST }
}
 
export function postHomeOrdersAddOneRequestSuccess (json) {
  return {
    type: POST_HOME_ORDERS_ADDONE_SUCCESS,
    payload: json
  }
}
export function postHomeOrdersAddOneRequestFailure (error) {
  return {
    type: POST_HOME_ORDERS__ADDONE_FAILURE,
    payload: error
  }
}

export function postPayMentList () {
  return dispatch => {

      .then(json => {

  }
}
