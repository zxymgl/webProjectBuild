import React, { PropTypes } from 'react'
import CountChangeInfoStyles from './CountChange.scss'
 
class CountChange extends React.Component {
  constructor (props) {
    super(props)
    this.state = {
      productNumber: 0
    }
    this._bind('_productNumberAdd', '_productNumberSubtract')
  }
  static propTypes = {
    productNumber: PropTypes.number
  }
  _bind (...methods) {
    methods.forEach((method) => {
      this[method] = this[method].bind(this)
    })
  }
  _productNumberAdd () {
    let number = this.state.productNumber
    this.setState({
      productNumber: number + 1
    })
  }
  _productNumberSubtract () {
    let number = this.state.productNumber
    if (number === 0) return
    this.setState({
      productNumber: number - 1
    })
  }
 
  render () {
    return (
      <div >
        <span className='glyphicon glyphicon-plus' aria-hidden='true' onClick={this._productNumberAdd}></span>
        <input type='text' value={this.state.productNumber} className={CountChangeInfoStyles.inputTextNumber} />
        <span className='glyphicon glyphicon-minus' aria-hidden='true' onClick={this._productNumberSubtract}></span>
      </div>
    )
  }
}
 
export default CountChange
