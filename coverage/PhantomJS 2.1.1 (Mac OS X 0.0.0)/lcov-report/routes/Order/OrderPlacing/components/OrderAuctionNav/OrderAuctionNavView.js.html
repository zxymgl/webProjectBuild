import React, { PropTypes } from 'react'
import classNames from 'classnames'
 
import currentStyle from './OrderAuctionNavView.scss'
import { IndexLink, Link } from 'react-router'
 
class OrderAuctionNavView extends React.Component {
  constructor (props) {
    super(props)
    this.state = {
    }
  }
  static propTypes = {
    switchMenuIndex: PropTypes.num
  }
  render () {
 
    const pages = [
      [ 'orderInfo', '提交订单' ],
      [ 'payment', '付款' ],
      [ 'orderConfirmation', '确认服务完成' ],
      [ 'orderReview', '评价' ]
    ]
    return (
      <div>
        <div className='col-lg-12 border-bottom'>
          <ul className='nav navbar-nav  m-l-xl' style={{ width: 800 }}>
            {pages.map((page, index) => (
              <li key={page[0]} className={ classNames(this.props.switchMenuIndex === index &&currentStyle.active ,'text-center')} style={{ width: 200 }} >
                <Link to={`/Orders/OrderPlacing/${page[0]}`}>{page[1]}</Link>
              </li>
            ))}
          </ul>
        </div>
      </div>
    )
  }
}
OrderAuctionNavView.defaultProps = {
  switchMenuIndex: 0
}
 
export default OrderAuctionNavView
