import React from 'react'
import { connect } from 'react-redux'
import { bindActionCreators } from 'redux'
import { Map } from 'immutable'
import classNames from 'classnames'
import styles from './Orders.scss'
import Chat from '../assets/u347.png'
import LeftNavigationBar from '../components/LeftNavigationBar'
import OrderList from '../routes/OrderList'
import MyReviews from '../routes/MyReviews'
import MyMessages from '../routes/MyMessages'
import { actions as myOrdersActions } from '../modules/myOrderList'
const actions = [
  myOrdersActions
]
function mapStateToProps (state) {
  return {
    messageList: state.orders.messagesListData,
    orderList: state.orders.ordersListData,
    reviewList: state.orders.reviewsListData
  }
}
function mapDispatchToProps (dispatch) {
  const creators = Map().merge(...actions).filter(value => typeof value === 'function').toObject()
  return {
    actions: bindActionCreators(creators, dispatch),
    dispatch
  }
}
class Orders extends React.Component {

    super(props)
    this.state = {
      current: 'one'
    }
    this._bind('_onMessages', '_onOrders', '_onReviews', '_renderViewer')
  }
  _bind (...methods) {
    methods.forEach((method) => {
      this[method] = this[method].bind(this)
    })
  }
  _renderViewer = () => {
    switch (this.state.current) {
      case 'one':
        return <OrderList />
      case 'two':
        return <MyReviews />
      case 'three':
        return <MyMessages />
    }
  }
  render () {
    let { children } = this.props
    return (
      <div className='col-md-12'>
        <div className='col-md-11'>
          <div id='wrapper' className='full-height full-width white-bg'>
            <LeftNavigationBar />
            <div id='page-wrapper' className='white-bg dashbard-1'>
              <br />
              <div>
                {children}
              </div>
            </div>
          </div>
        </div>
        <div className='col-md-1'>
          <img src={Chat} className={classNames(styles.chatbg, 'pull-right')} onClick={this._onChat} />
        </div>
      </div>
    )
  }
  componentDidMount () {
    this.props.actions.getOrderList()
    this.props.actions.getMessageList()
    this.props.actions.getReviewList()
  }
  _onOrders () {
    this.setState({current: 'one'})
  }
  _onMessages () {
    this.setState({current: 'two'})
  }
  _onReviews () {
    this.setState({current: 'three'})
  }
  _onChat () {
  }
}

  actions: React.PropTypes.object.isRequired,
  messageList: React.PropTypes.object,
  orderList: React.PropTypes.object,
  reviewList: React.PropTypes.object,
  children: React.PropTypes.element
}
export default connect(mapStateToProps, mapDispatchToProps)(Orders)
