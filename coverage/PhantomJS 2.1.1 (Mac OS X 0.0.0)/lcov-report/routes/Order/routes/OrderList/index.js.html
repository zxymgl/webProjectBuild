import { injectReducer } from 'store/reducers'
import createOrderListRoute from './routes/index'
export default (store) => {
  let childRoute = createOrderListRoute(store)
  return {
    path: 'OrderList',
    getComponent (nextState, cb) {
      require.ensure([], (require) => {
        const OrderList = require('./containers/OrderList').default
        const reducer = require('./modules/myOrderList').default
        injectReducer(store, {key: 'orderlist', reducer})
        cb(null, OrderList)
      }, 'OrderList')
    },
    childRoutes: childRoute.childRoutes
  }
}
