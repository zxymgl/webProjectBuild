import React from 'react'
import { Link } from 'react-router'
import styles from './OrderStatusNavigationBar.scss'
class OrderStatusNavigationBar extends React.Component {
  constructor (props) {
    super(props)
    this.state = {
      type: 'one',
      SwitchMenuIndex: 1
    }
    this._handleSwitchMenuIndex = this._handleSwitchMenuIndex.bind(this)
  }
  _bind (...methods) {
    methods.forEach((method) => {
      this[method] = this[method].bind(this)
    })
  }
  render () {
    return (
      <div className='col-lg-12 border-bottom'>
        <ul className='nav navbar-nav '>
          <li className={this.state.SwitchMenuIndex === 1 && styles.active} onClick={() => {this._handleSwitchMenuIndex(1) }}>
            <Link to='/orders/OrderList/AllOrders' onClick={this._onCheckout.bind(this, 'one')}>
              <h4>全部订单</h4>
            </Link>
          </li>
          <li className={this.state.SwitchMenuIndex === 2 && styles.active} onClick={() => {this._handleSwitchMenuIndex(2) }}>
            <Link to='/orders/OrderList/WaitingForPaymentOrders' onClick={this._onCheckout.bind(this, 'two')}>
              <h4>待付款</h4>
            </Link>
          </li>
          <li className={this.state.SwitchMenuIndex === 3 && styles.active} onClick={() => {this._handleSwitchMenuIndex(3) }}>
            <Link to='/orders/OrderList/WaitingForHandleOrders' onClick={this._onCheckout.bind(this, 'three')}>
              <h4>待受理</h4>
            </Link>
          </li>
          <li className={this.state.SwitchMenuIndex === 4 && styles.active} onClick={() => {this._handleSwitchMenuIndex(4) }}>
            <Link to='/orders/OrderList/WaitingForConfirmOrders' onClick={this._onCheckout.bind(this, 'four')}>
              <h4>待确认</h4>
            </Link>
          </li>
          <li className={this.state.SwitchMenuIndex === 5 && styles.active} onClick={() => {this._handleSwitchMenuIndex(5) }}>
            <Link to='/orders/OrderList/WaitingForReviewOrders' onClick={this._onCheckout.bind(this, 'five')}>
              <h4>待评价</h4>
            </Link>
          </li>
        </ul>
      </div>
    )
  }
  _handleSwitchMenuIndex (menuIndex) {
    this.setState({
      SwitchMenuIndex: menuIndex
    })
  }
  _onCheckout (e) {
    this.setState({type: e})
  }
}
export default OrderStatusNavigationBar
