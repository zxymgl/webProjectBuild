import Immutable, { Record, List } from 'immutable'
import apiFactory from 'api/apiFactory'
const api = apiFactory()
import listData from './OrderListData'
import reviewData from './ReviewListData'
import messageData from './MessageListData'
/**
 * 获取我的订单列表数据
 * @type {string}
 */
export const GET_ORDERLIST_REQUEST = 'GET_ORDERLIST_REQUEST'
export const GET_ORDERLIST_SUCCESS = 'GET_ORDERLIST_SUCCESS'
export const GET_ORDERLIST_FAILURE = 'GET_ORDERLIST_FAILURE'
/**
 * 获取我的订单列表
 * @returns {{type: string}}
 */
export function getOrderListRequest () {
  return {type: GET_ORDERLIST_REQUEST}
}
export function getOrderListSuccess (json) {
  return {
    type: GET_ORDERLIST_SUCCESS,
    payload: json
  }
}
export function getOrderListFailure (error) {
  return {
    type: GET_ORDERLIST_FAILURE,
    payload: error
  }
}
export function getOrderList () {
  return dispatch => {
    dispatch(getOrderListRequest())
    return api.products.categories.getProductCategoryList('CTX2000')
      .then(json => {
        dispatch(getOrderListSuccess(json))
      }).catch(error => {
        dispatch(getOrderListFailure(error))
      })
  }
}

export const actions = {
  getOrderList,
  getMessageList,
  getReviewList
}
let InitialState = new Record({
  isLauncher: false,
  ordersListData: new List(),
  isLoading: false,
  messagesListData: new List(),
  isFetching: false,
  reviewsListData: new List()
})
const initialState = new InitialState()
export default function OrderListReducer (state = initialState, action) {
  if (!(state instanceof InitialState)) return initialState.merge(state)
  switch (action.type) {
    case GET_ORDERLIST_REQUEST:
      return state.set('isLauncher', false)
    case GET_ORDERLIST_SUCCESS:
      var orderList = Immutable.fromJS(listData)
      return state.set('isLauncher', false).set('ordersListData', orderList)
    case GET_ORDERLIST_FAILURE:
      return state.set('isLauncher', false)
    case GET_MESSAGELIST_REQUEST:
      return state.set('isLoading', false)
    case GET_MESSAGELIST_SUCCESS:
      var messageList = Immutable.fromJS(messageData)
      return state.set('isLoading', false).set('messagesListData', messageList)
    case GET_MESSAGELIST_FAILURE:
      return state.set('isLoading', false)
    case GET_REVIEWLIST_REQUEST:
      return state.set('isFetching', false)
    case GET_REVIEWLIST_SUCCESS:
      var reviewList = Immutable.fromJS(reviewData)
      return state.set('isFetching', false).set('reviewsListData', reviewList)
    case GET_REVIEWLIST_FAILURE:
      return state.set('isFetching', false)
  }
  return state
}
