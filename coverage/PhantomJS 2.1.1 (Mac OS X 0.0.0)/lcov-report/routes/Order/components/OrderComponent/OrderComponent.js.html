import React from 'react'
import ListGroup from 'react-bootstrap/lib/ListGroup'
import OrderListCell from '../OrderListCell'
class OrderComponent extends React.Component {
  constructor (props) {
    super(props)
    this.state = { }
  }
  render () {
    return (
      <div>
        <ListGroup>
          {this.props.data.map((v, i) => {
            if (this.props.status === 'all') {
              return (
                <OrderListCell key={i} type={this.props.type} celltype={this.props.celltype} data={v} />
              )
            } else if (this.props.status === 'waitpay') {
              if (v.get('state') === 2) {
                return (
                  <OrderListCell key={i} type={this.props.type} celltype={this.props.celltype} data={v} />
                )
              }
            } else if (this.props.status === 'waithandle') {
              if (v.get('state') === 3) {
                return (
                  <OrderListCell key={i} type={this.props.type} celltype={this.props.celltype} data={v} />
                )
              }
            } else if (this.props.status === 'waitconfirm') {
              if (v.get('state') === 4) {
                return (
                  <OrderListCell key={i} type={this.props.type} celltype={this.props.celltype} data={v} />
                )
              }
            } else if (this.props.status === 'waitreview') {
              if (v.get('state') === 5) {
                return (
                  <OrderListCell key={i} type={this.props.type} celltype={this.props.celltype} data={v} />
                )
              }
            } else if (this.props.status === 'other') {
              return (
                <OrderListCell key={i} type={this.props.type} celltype={this.props.celltype} data={v} />
              )
            }
          })}
        </ListGroup>
      </div>
    )
  }
}
OrderComponent.propTypes = {
  data: React.PropTypes.object,
  type: React.PropTypes.string,
  celltype: React.PropTypes.string,
  status: React.PropTypes.string
}
export default OrderComponent
