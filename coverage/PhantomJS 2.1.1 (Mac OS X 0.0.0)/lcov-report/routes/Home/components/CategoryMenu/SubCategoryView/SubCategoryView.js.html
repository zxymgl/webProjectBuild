import React from 'react'
import { List } from 'immutable'
 
class SubCategoryView extends React.Component {
  constructor (props) {
    super(props)
    this.state = {
      activeMenuIndex: -1
    }
  }
  render () {
    return (
      <div className='ibox border-bottom border-left-right m-b-xs'>
        <div className='ibox-title'>
          <h4>{this.props.data.get('productCategory').get('categoryName')}</h4>
        </div>
        <div className='ibox-content'>
          <div className='row'>
            {this.props.data.get('children') && this.props.data.get('children').map(function (subMenu, index) {
              let productCategory = subMenu.get('productCategory')
              return (
                <div className='col-md-4 border-left m-b-sm' key={index}>
                  <a>{productCategory.get('categoryName')}</a>
                </div>
              )
            }.bind(this))}
          </div>
        </div>
      </div>
 
    )
  }
}
SubCategoryView.propTypes = {
  data: React.PropTypes.object
}
 
export default SubCategoryView
