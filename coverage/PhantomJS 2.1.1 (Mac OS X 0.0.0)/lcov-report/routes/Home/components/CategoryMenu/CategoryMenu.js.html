import React from 'react'
import { List } from 'immutable'
import ReactMenuAim from 'components/ReactMenuAim'
import styles from './CategoryMenu.scss'
 
import SubCategoryView from './SubCategoryView/SubCategoryView'
 
class CategoryMenu extends ReactMenuAim {
  constructor (props) {
    super(props)
    this.state = {
      activeMenuIndex: -1
    }
    // this.handleMouseLeaveMenu = this.handleMouseLeaveMenu.bind(this)
    this._bind('handleMouseLeaveMenu', 'resetActiveMenuIndex')
  }
 
  _bind (...methods) {
    methods.forEach((method) => {
      this[method] = this[method].bind(this)
    })
  }
 
  componentWillMount () {
    this.initMenuAim({
      submenuDirection: this.props.submenuDirection,
      menuSelector: '.' + styles['menu'],
      delay: 300,
      tolerance: 75
    })
  }
 
  handleSwitchMenuIndex (index) {
    this.setState({
      activeMenuIndex: index
    })
  }
 
  resetActiveMenuIndex () {
    this.setState({
      activeMenuIndex: -1
    })
  }
 
  render () {
    let self = this
    var subMenuStyle = {}
    if (this.props.submenuDirection === styles['below']) {
      subMenuStyle.left = this.state.activeMenuIndex * 140
    }
 
    return (
      <div className='row m-md' onMouseLeave={this.resetActiveMenuIndex}>
        <div className='col-md-2 no-padding' >
          <ul className={styles['menu']} onMouseLeave={this.handleMouseLeaveMenu}>
            {this.props.data.map(function (menu, index) {
              var className = styles['menu-item']
              if (index === self.state.activeMenuIndex) {
                className += ' ' + styles['active']
              }
              var productCategory = menu.get('productCategory')
              return (
                <li className={className} key={index}
                    onMouseEnter={self.handleMouseEnterRow.bind(self, index, self.handleSwitchMenuIndex)}>
                  <a className="text-center" href={productCategory.get('productCategoryId')}>
                    <span >{productCategory.get('categoryName')}</span>
                  </a>
                </li>
              )
            })}
          </ul>
        </div>
        <div className='col-md-5 no-padding'>
          {this.state.activeMenuIndex >= 0 && this.props.data.get(this.state.activeMenuIndex).get('children')
            ? (
            <ul className='list-unstyled' style={{border: '1px solid #337ab7'}}>
              {this.props.data.get(this.state.activeMenuIndex).get('children').map(function (subMenu, index) {
                return (
                  <SubCategoryView data={subMenu} key={index} />
                )
              })}
            </ul>)
            : null
          }
        </div>
      </div>
    )
  }
}
 
CategoryMenu.defaultProps = {
  submenuDirection: styles['right']
}
 
CategoryMenu.propTypes = {
  submenuDirection: React.PropTypes.string,
  data: React.PropTypes.object
}
 
export default CategoryMenu
