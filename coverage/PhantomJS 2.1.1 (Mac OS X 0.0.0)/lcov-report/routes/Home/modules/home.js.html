import Immutable, { Record, List } from 'immutable'
 
import apiFactory from 'api/apiFactory'
 
const api = apiFactory()
 
  // 获取首页目录的产品种类
export const GET_HOME_PRODUCT_CATEGORIES_REQUEST = 'GET_HOME_PRODUCT_CATEGORIES_REQUEST'
export const GET_HOME_PRODUCT_CATEGORIES_SUCCESS = 'GET_HOME_PRODUCT_CATEGORIES_SUCCESS'
export const GET_HOME_PRODUCT_CATEGORIES_FAILURE = 'GET_HOME_PRODUCT_CATEGORIES_FAILURE'
 
export function getHomeProductCategoriesRequest () {
  return { type: GET_HOME_PRODUCT_CATEGORIES_REQUEST }
}
 
export function getHomeProductCategoriesSuccess (json) {
  return {
    type: GET_HOME_PRODUCT_CATEGORIES_SUCCESS,
    payload: json
  }
}
 
export function getHomeProductCategoriesFailure (error) {
  return {
    type: GET_HOME_PRODUCT_CATEGORIES_FAILURE,
    payload: error
  }
}
 
export function getHomeProductCategories () {
  return dispatch => {
    dispatch(getHomeProductCategoriesRequest())
    return api.products.categories.one.childCategories.get('ZC_BrowseRoot', '3')
      .then(json => {
        dispatch(getHomeProductCategoriesSuccess(json))
      }).catch(error => {
        dispatch(getHomeProductCategoriesFailure(error))
      })
  }
}
 
export const actions = {
  getHomeProductCategories
}
 
let InitialState = Record({
  isFetching: false,
  productCategoryList: new List()
})
 
const initialState = new InitialState()
 
export default function homeReducer (state = initialState, action) {
  if (!(state instanceof InitialState)) return initialState.merge(state)
 
  switch (action.type) {
    case GET_HOME_PRODUCT_CATEGORIES_REQUEST:
      return state.set('isFetching', true)
 
    case GET_HOME_PRODUCT_CATEGORIES_SUCCESS:
      if (!Immutable.fromJS(action.payload)) return null
      var pcMap = Immutable.fromJS(action.payload)
      return state.set('isFetching', false)
                  .set('productCategoryList', pcMap.get('productCategoryList'))
 
    case GET_HOME_PRODUCT_CATEGORIES_FAILURE:
      return state.set('isFetching', false)
  }
  return state
}
