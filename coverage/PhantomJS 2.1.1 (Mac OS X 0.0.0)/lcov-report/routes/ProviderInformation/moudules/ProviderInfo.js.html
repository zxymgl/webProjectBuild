import Immutable, { Record, Map } from 'immutable'
 
import apiFactory from 'api/apiFactory'
 
const api = apiFactory()
 
// 获取person信息
export const GET_PARTIES_PERSONS_ONE_REQUEST = 'GET_PARTIES_PERSONS_ONE_REQUEST'
export const GET_PARTIES_PERSONS_ONE_SUCCESS = 'GET_PARTIES_PERSONS_ONE_SUCCESS'
export const GET_PARTIES_PERSONS_ONE_FAILURE = 'GET_PARTIES_PERSONS_ONE_FAILURE'
 
export function getPartiesPersonsOneRequest () {
  return { type: GET_PARTIES_PERSONS_ONE_REQUEST }
}
 
export function getPartiesPersonsOneSuccess (json) {
  return {
    type: GET_PARTIES_PERSONS_ONE_SUCCESS,
    payload: json
  }
}
export function getPartiesPersons_One_Failure (error) {
  return {
    type: GET_PARTIES_PERSONS_ONE_FAILURE,
    payload: error
  }
}
export function getPartiesPersonsOne (partyId) {
  return dispatch => {
    dispatch(getPartiesPersonsOneRequest())
    return api.parties.persons.one.get(partyId)
      .then(json => {
        dispatch(getPartiesPersonsOneSuccess(json))
      }).catch(error => {
        dispatch(getPartiesPersons_One_Failure(error))
      })
  }
}

let InitialState = Record({
  isFetching: false,
  personInfo: new Map(),
  contactTele: new Map()
})
 
const initialState = new InitialState()
 
export default function providerInfoReducer (state = initialState, action) {
  if (!(state instanceof InitialState)) return initialState.merge(state)
  switch (action.type) {
    case GET_PARTIES_PERSONS_ONE_REQUEST:
      return state.set('isFetching', true)
    case GET_PARTIES_PERSONS_ONE_SUCCESS:
      if (!Immutable.fromJS(action.payload)) return null
      let pcMap = Immutable.fromJS(action.payload)
      return state.set('isFetching', false)
        .set('personInfo', pcMap.get('person'))
    case GET_PARTIES_PERSONS_ONE_FAILURE:
      return state.set('isFetching', false)
  }
  switch (action.type) {
    case GET_PARTIES_CONTACTMECHS_TELECOMNUMBERS_ONE_REQUEST:
      return state.set('isFetching', true)
    case GET_PARTIES_CONTACTMECHS_TELECOMNUMBERS_ONE_SUCCESS:
      if (!Immutable.fromJS(action.payload)) return null
      let pcMap = Immutable.fromJS(action.payload)
      return state.set('isFetching', false)
        .set('personInfo', pcMap.get('person'))
    case GET_PARTIES_CONTACTMECHS_TELECOMNUMBERS_ONE_FAILURE:
      return state.set('isFetching', false)
  }
  return state
}
