import styles from './ProductReviews.scss'

    super(props)
    this.state = {
    }
  }
  render () {
    return (
      <div>
        <ListGroup>
          {this.props.reviews.map(function (review, index) {
            return (
              <ListGroupItem key={index}>
                <div className={styles.reviewlbg}>
                  <p>{review.get('productReview')}</p>
                  <div>
                    <span>{review.get('postedDateTime')}</span>
                  </div>
                </div>
              </ListGroupItem>
            )
          })}

    )
  }
}
