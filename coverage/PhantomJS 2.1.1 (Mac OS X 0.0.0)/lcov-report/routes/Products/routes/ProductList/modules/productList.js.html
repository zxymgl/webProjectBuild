const getProducts = (categoryQuery, priceQuery, pagination) => dispatch => {
  dispatch({ type: GET_CATEGORY_PRODUCT_LIST_REQUEST })
  let productCategoryId = 'ZC_BrowseRoot'
  if (categoryQuery) {
    if (categoryQuery.level_1) productCategoryId = categoryQuery.level_1
    if (categoryQuery.level_2) productCategoryId = categoryQuery.level_2
  }
 
  return api.products.categories.one.products.get(productCategoryId, { level: 10 }, pagination)
  .then(json => {
    dispatch({ type: GET_CATEGORY_PRODUCT_LIST_SUCCESS, payload: json })
  }).catch(error => {
    dispatch({ type: GET_CATEGORY_PRODUCT_LIST_FAILURE, payload: error })
  })
}
