import React from 'react'
import ImmutablePropTypes from 'react-immutable-proptypes'
 
import styles from './ProductCategorySearchCriterias.scss'
 
const CategoryLevel = ({ labelText, categoryList, selectedId, categoryClickCallBack }) => (
  <div>
    <div className={styles['category-name']}>{labelText}</div>
    <div className={styles['category-list']}>
      <ul className={styles['facet-menu']}>
        {categoryList.map((it, index) => (
          <li key={index} className={selectedId === it.productCategory.productCategoryId ? 'active' : ''}>
            <a onClick={categoryClickCallBack(it.productCategory.productCategoryId)}>
              {it.productCategory.categoryName}</a>
          </li>
        ))
        }
      </ul>
    </div>
  </div>
)
 
class ProductCategorySearchCriterias extends React.Component {
  constructor (props) {
    super(props)
    this.state = {
      level1SelectedId: null,
      level2SelectedId: null,
      level3SelectedId: null,
      level1CategoryList: this.props.productCategoryList,
      level2CategoryList: null,
      level3CategoryList: null
    }
    this.handleLevel1CategoryClick = this.handleLevel1CategoryClick.bind(this)
    this.handleLevel2CategoryClick = this.handleLevel2CategoryClick.bind(this)
    this.handleLevel3CategoryClick = this.handleLevel3CategoryClick.bind(this)
  }
 
  handleLevel1CategoryClick (productCategoryId) {
    return () => {
      this.setState({
        'level1SelectedId': productCategoryId,
        'level2SelectedId': null,
        'level3SelectedId': null,
        'level2CategoryList': this.state.level1CategoryList.find(it => it.productCategory.productCategoryId === productCategoryId).get('children'),
        'level3CategoryList': null
      })
      this.props.categoryChangedCallBack(productCategoryId)
    }
  }
 
  handleLevel2CategoryClick (productCategoryId) {
    return () => {
      this.setState({
        'level2SelectedId': productCategoryId,
        'level3SelectedId': null,
        'level3CategoryList': this.state.level2CategoryList.find(it => it.productCategory.productCategoryId === productCategoryId).get('children')
      })
      this.props.categoryChangedCallBack(productCategoryId)
    }
  }
 
  handleLevel3CategoryClick (productCategoryId) {
    return () => {
      this.setState({'level3SelectedId': productCategoryId})
      this.props.categoryChangedCallBack(productCategoryId)
    }
  }
 
  componentWillReceiveProps (nextProps) {
    this.setState({
      level1SelectedId: null,
      level2SelectedId: null,
      level3SelectedId: null,
      level1CategoryList: nextProps.productCategoryList,
      level2CategoryList: null,
      level3CategoryList: null
    })
  }
 
  render () {
    return (
      <div>
        <CategoryLevel labelText='分类' categoryList={this.state.level1CategoryList}
                       selectedId={this.state.level1SelectedId} categoryClickCallBack={this.handleLevel1CategoryClick} />
        {this.state.level1SelectedId
          ? <CategoryLevel labelText='二类' categoryList={this.state.level2CategoryList}
                           selectedId={this.state.level2SelectedId} categoryClickCallBack={this.handleLevel2CategoryClick} />
          : null
        }
        {this.state.level2SelectedId
          ? <CategoryLevel labelText='三类' categoryList={this.state.level3CategoryList}
                           selectedId={this.state.level3SelectedId} categoryClickCallBack={this.handleLevel3CategoryClick} />
          : null
        }
      </div>
    )
  }
}
 
ProductCategorySearchCriterias.propTypes = {
  productCategoryList: ImmutablePropTypes.list,
  categoryChangedCallBack: React.PropTypes.func
}
export default ProductCategorySearchCriterias
